    # File lib/dLinkedList/dLinkedList.rb, line 272
def pop()
        
        raise RuntimeError, "[List.pop]: No se puede extraer elementos de una lista vacía" if empty?()
        
        nodo = @head
        if @head.equal?(@tail)
                @head = nil
                @tail = nil
        else
                @head = @head.next_node
                @head.prev_node = nil
        end
        
        @size -= 1
        
        return nodo.value
        
end

def push(ref)
        if empty?()
                return push_empty(ref)
        else
                nodo = Node.new(ref, nil, @tail)
                @tail.next_node = nodo
                @tail = nodo
                @size += 1
                return self
        end
end
